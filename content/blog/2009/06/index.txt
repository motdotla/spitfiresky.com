#articles
  - articles = @pages.find(:all, :in_directory => @page.dir, :recursive => true, :sort_by => "created_at", :reverse => true, :blog_post => true)

  - articles.each do |article|
    %li
      %span= article.created_at.strftime('%d %b %Y')
      &raquo;
      = link_to_page(article)
